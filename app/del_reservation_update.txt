#!/usr/bin/python3 
import psycopg2, cgi 
import login
form = cgi.FieldStorage()

# Get values from the form in previous page 
cni = form.getvalue('cni')
id_sailor = form.getvalue('id_sailor')
iso_code_sailor = form.getvalue('iso_code_sailor')
iso_code_boat = form.getvalue('iso_code_boat')
start_date = form.getvalue('start_date')
end_date = form.getvalue('end_date')

print('Content-type:text/html\n\n')
print('<html>')
print('<head>')
print('<title>Boat Management System</title>')
print('</head>')
print('<body>')

connection = None
try:
        # Creating connection
        connection = psycopg2.connect(login.credentials)
        cursor = connection.cursor()

        # Making query
        sql_trip = "DELETE FROM trip WHERE cni=cni AND iso_code_boat=iso_code_boat AND id_sailor=id_sailor AND iso_code_sailor=iso_code_sailor AND start_date=start_date AND end_date=end_date";
        sql_reservation = "DELETE FROM reservation WHERE cni=cni AND iso_code_boat=iso_code_boat AND id_sailor=id_sailor AND iso_code_sailor=iso_code_sailor AND start_date=start_date AND end_date=end_date";
        data = {'cni':cni,'iso_code_boat':iso_code_boat,'id_sailor':id_sailor,'iso_code_sailor':iso_code_sailor,'start_date':start_date,'end_date':end_date}


        # The string has the {}, the variables inside format() will replace the {}
        print('<p>{}</p>'.format(sql_trip % data))
        print('<p>{}</p>'.format(sql_reservation % data))
        # Feed the data to the SQL query as follows to avoid SQL injection
        cursor.execute(sql_trip, data)
        cursor.execute(sql_reservation, data)

        # Commit the update (without this step the database will not change)
        connection.commit()

        # Closing connection
        cursor.close()

except Exception as e:
        # Print errors on the webpage if they occur
        print('<h1>An error occurred.</h1>')
        print('<p>{}</p>'.format(e))
finally:
        if connection is not None:
                connection.close()
print('</body>')
print('</html>')
